function getRealFloor(n) {
  if (n <= 0) {
    // Если этаж нулевой или отрицательный, то он остается таким же
    return n;
  } else if (n < 13) {
    // Если этаж меньше 13, то он остается на том же уровне
    return n - 1;
  } else {
    // Если этаж больше или равен 13, то он перемещается на один уровень вниз
    return n - 2;
  }
}
